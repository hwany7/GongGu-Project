/* member */
create table member_gg(
    member_id number primary key,
    member_email varchar2(50) unique not null,
    password varchar2(20) not null,
    nickname varchar2(30) unique not null,
    zipcode varchar2(6) not null,
    address1 varchar2(150) not null,
    address2 varchar2(150) not null,
    tel varchar2(20) not null,
    member_status varchar2(10) default 'active',			-- active, suspended, dormant, inactive Ȱ��, Ȱ������, �޸�, Ż��
    join_date date default sysdate,
    gender varchar2(10),	
    birth_date date,
    visit_path varchar2(20)
);
-- auto_increment�ؼ� pk�� �����ϱ�
create sequence member_gg_sequence start with 1 increment by 1;
create or replace trigger member_gg_on_insert 
    before insert on member_gg 
    for each row

    select member_gg_sequence.nextval 
    into :new.member_id 
    from dual;
end;

    into :new.reply_num
    from dual;
end;
